064/**
065 * An {@link AbstractContainerRequestHandlingResponseWriter}
066 * implemented in terms of {@link HttpRequest}s, {@link HttpResponse}s
067 * and {@link
068 * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}s.
069 *
070 * @author <a href="https://about.me/lairdnelson"
071 * target="_parent">Laird Nelson</a>
072 *
073 * @see AbstractContainerRequestHandlingResponseWriter
074 *
075 * @see ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
076 *
077 * @see #writeStatusAndHeaders(long, ContainerResponse)
078 *
079 * @see #createOutputStream(long, ContainerResponse)
080 */
081public final class HttpContainerRequestHandlingResponseWriter extends AbstractContainerRequestHandlingResponseWriter<HttpContent> {

176  /**
177   * Writes the status and headers portion of the response present in
178   * the supplied {@link ContainerResponse} and returns {@code true}
179   * if further output is forthcoming.
180   *
181   * <p>This implementation writes an instance of either {@link
182   * DefaultHttpResponse} or {@link DefaultFullHttpResponse}.</p>
183   *
184   * @param contentLength the content length as determined by the
185   * logic encapsulated by the {@link
186   * ApplicationHandler#handle(ContainerRequest)} method; a value less
187   * than zero indicates an unknown content length
188   *
189   * @param containerResponse the {@link ContainerResponse} containing
190   * status and headers information; must not be {@code null}
191   *
192   * @return {@code true} if the {@link #createOutputStream(long,
193   * ContainerResponse)} method should be invoked, <em>i.e.</em> if
194   * further output is forthcoming
195   *
196   * @exception NullPointerException if {@code containerResponse} is
197   * {@code null}
198   *
199   * @exception IllegalArgumentException if the supplied {@link
200   * ContainerResponse} returns {@code null} from its {@link
201   * ContainerResponse#getRequestContext()} method or if that {@link
202   * ContainerRequest} returns something that is not an {@link
203   * HttpRequest} from invoking {@link
204   * ContainerRequest#getProperty(String)
205   * getProperty("io.netty.handler.codec.http.HttpRequest")} on it
206   *
207   * @see ApplicationHandler#handle(ContainerRequest)
208   *
209   * @see #createOutputStream(long, ContainerResponse)
210   */
211  @Override
212  protected final boolean writeStatusAndHeaders(final long contentLength,
213                                                final ContainerResponse containerResponse) {
214    final ContainerRequest containerRequest = containerResponse.getRequestContext();
215    if (containerRequest == null) {
216      throw new IllegalArgumentException("containerResponse.getRequestContext() == null");
217    }
218
219    final HttpRequest httpRequest;
220    final Object httpRequestValue = containerRequest.getProperty(HttpRequest.class.getName());
221    if (!(httpRequestValue instanceof HttpRequest)) {
222      throw new IllegalArgumentException("containerResponse; !(containerResponse.getRequestContext().getProperty(\"" +
223                                         HttpRequest.class.getName() +
224                                         "\") instanceof HttpRequest): " + httpRequestValue);
225    } else {
226      httpRequest = (HttpRequest)httpRequestValue;
227    }
228
229    final HttpVersion httpVersion = httpRequest.protocolVersion();
230    this.httpVersion = httpVersion;
231
232    final HttpResponseStatus status;
233    final StatusType responseStatusType = containerResponse.getStatusInfo();
234    if (responseStatusType == null) {
235      status = HttpResponseStatus.valueOf(containerResponse.getStatus());
236    } else {
237      final String reasonPhrase = responseStatusType.getReasonPhrase();
238      if (reasonPhrase == null) {
239        status = HttpResponseStatus.valueOf(containerResponse.getStatus());
240      } else {
241        status = HttpResponseStatus.valueOf(containerResponse.getStatus(), reasonPhrase);
242      }
243    }
244
245    final HttpMessage httpResponse;
246    final boolean needsOutputStream;
247    if (contentLength < 0L) {
248      needsOutputStream = !HttpMethod.HEAD.equalsIgnoreCase(containerRequest.getMethod());
249      httpResponse = new DefaultHttpResponse(httpVersion, status);
250      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
251      HttpUtil.setTransferEncodingChunked(httpResponse, true);
252    } else if (contentLength == 0L) {
253      needsOutputStream = false;
254      httpResponse = new DefaultFullHttpResponse(httpVersion, status);
255      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
256      HttpUtil.setContentLength(httpResponse, 0L);
257    } else {
258      needsOutputStream = !HttpMethod.HEAD.equalsIgnoreCase(containerRequest.getMethod());
259      httpResponse = new DefaultHttpResponse(httpVersion, status);
260      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
261      HttpUtil.setContentLength(httpResponse, contentLength);
262    }
263    if (HttpUtil.isKeepAlive(httpRequest)) {
264      HttpUtil.setKeepAlive(httpResponse, true);
265    }
266
267    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
268
269    final ChannelPromise channelPromise = channelHandlerContext.newPromise();
270    assert channelPromise != null;
271    channelPromise.addListener(listener);
272
273    // Remember that
274    // AbstractContainerRequestHandlingResponseWriter#channelReadComplete(ChannelHandlerContext)
275    // will call ChannelHandlerContext#flush() in all cases.
276    channelHandlerContext.write(httpResponse, channelPromise);
277
278    return needsOutputStream;
279  }

281  /**
282   * Creates and returns a new {@link
283   * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}.
284   *
285   * <p>This method never returns {@code null}.</p>
286   *
287   * @param contentLength the content length as determined by the
288   * logic encapsulated by the {@link
289   * ApplicationHandler#handle(ContainerRequest)} method; must not be
290   * {@code 0L}
291   *
292   * @param containerResponse a {@link ContainerResponse} for which an
293   * {@link OutputStream} is being created and returned; must not be
294   * {@code null}; ignored by this implementation
295   *
296   * @return a new {@link
297   * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}
298   *
299   * @exception NullPointerException if {@code containerResponse} is
300   * {@code null} or if {@link #getChannelHandlerContext()} returns
301   * {@code null}
302   *
303   * @exception IllegalArgumentException if {@code contentLength} is
304   * {@code 0L}
305   *
306   * @see ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
307   */
308  @Override
309  protected final AbstractChannelOutboundInvokingOutputStream<? extends HttpContent> createOutputStream(final long contentLength,
310                                                                                                        final ContainerResponse containerResponse) {
311    if (contentLength == 0L) {
312      throw new IllegalArgumentException("contentLength == 0L");
313    }
314    return new ByteBufBackedChannelOutboundInvokingHttpContentOutputStream(this.getChannelHandlerContext(),
315                                                                           this.getFlushThreshold(),
316                                                                           false,
317                                                                           this.getByteBufCreator()) {
318      @Override
319      protected final ChannelPromise newPromise() {
320        final ChannelPromise returnValue = super.newPromise();
321        if (returnValue != null && !returnValue.isVoid()) {
322          returnValue.addListener(listener);
323        }
324        return returnValue;
325      }
326    };
327  }
328
329  /**
330   * Overrides {@link
331   * AbstractContainerRequestHandlingResponseWriter#commit()} to still
332   * effectively do nothing, but clean up some internal state.
333   *
334   * @see AbstractContainerRequestHandlingResponseWriter#commit()
335   */
336  @Override
337  public final void commit() {
338    this.httpVersion = null;
339    super.commit();
340  }
341
342  /**
343   * Writes an appropriate failure message using the return value of
344   * the {@link #getChannelHandlerContext()} method.
345   *
346   * @param failureCause the {@link Throwable} responsible for this
347   * method's invocation; may be {@code null} in pathological cases;
348   * ignored by this implementation
349   *
350   * @exception NullPointerException if {@link
351   * #getChannelHandlerContext()} returns {@code null}
352   */
353  @Override
354  protected final void writeFailureMessage(final Throwable failureCause) {
355    final HttpVersion httpVersion = this.httpVersion;
356    this.httpVersion = null;
357
358    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
359
360    final ChannelPromise channelPromise = channelHandlerContext.newPromise();
361    assert channelPromise != null;
362    channelPromise.addListener(listener);
363
364    final HttpMessage failureMessage = new DefaultFullHttpResponse(httpVersion, HttpResponseStatus.INTERNAL_SERVER_ERROR);
365    HttpUtil.setContentLength(failureMessage, 0L);
366
367    channelHandlerContext.write(failureMessage, channelPromise);
368  }

376  private static final void copyHeaders(final Map<? extends String, ? extends List<String>> headersSource,
377                                        final HttpHeaders nettyHeaders) {
378    AbstractContainerRequestHandlingResponseWriter.copyHeaders(headersSource, UnaryOperator.identity(), nettyHeaders::add);
379  }
