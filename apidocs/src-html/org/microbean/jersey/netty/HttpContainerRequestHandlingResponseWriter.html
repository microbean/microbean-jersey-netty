056/**
057 * An {@link AbstractContainerRequestHandlingResponseWriter}
058 * implemented in terms of {@link HttpRequest}s, {@link HttpResponse}s
059 * and {@link
060 * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}s.
061 *
062 * @author <a href="https://about.me/lairdnelson"
063 * target="_parent">Laird Nelson</a>
064 *
065 * @see AbstractContainerRequestHandlingResponseWriter
066 *
067 * @see ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
068 *
069 * @see #writeStatusAndHeaders(long, ContainerResponse)
070 *
071 * @see #createOutputStream(long, ContainerResponse)
072 */
073public final class HttpContainerRequestHandlingResponseWriter extends AbstractContainerRequestHandlingResponseWriter<HttpContent> {

168  /**
169   * Writes the status and headers portion of the response present in
170   * the supplied {@link ContainerResponse} and returns {@code true}
171   * if further output is forthcoming.
172   *
173   * <p>This implementation writes an instance of either {@link
174   * DefaultHttpResponse} or {@link DefaultFullHttpResponse}.</p>
175   *
176   * @param contentLength the content length as determined by the
177   * logic encapsulated by the {@link
178   * ApplicationHandler#handle(ContainerRequest)} method; a value less
179   * than zero indicates an unknown content length
180   *
181   * @param containerResponse the {@link ContainerResponse} containing
182   * status and headers information; must not be {@code null}
183   *
184   * @return {@code true} if the {@link #createOutputStream(long,
185   * ContainerResponse)} method should be invoked, <em>i.e.</em> if
186   * further output is forthcoming
187   *
188   * @exception NullPointerException if {@code containerResponse} is
189   * {@code null}
190   *
191   * @exception IllegalArgumentException if the supplied {@link
192   * ContainerResponse} returns {@code null} from its {@link
193   * ContainerResponse#getRequestContext()} method or if that {@link
194   * ContainerRequest} returns something that is not an {@link
195   * HttpRequest} from invoking {@link
196   * ContainerRequest#getProperty(String)
197   * getProperty("io.netty.handler.codec.http.HttpRequest")} on it
198   *
199   * @see ApplicationHandler#handle(ContainerRequest)
200   *
201   * @see #createOutputStream(long, ContainerResponse)
202   */
203  @Override
204  protected final boolean writeStatusAndHeaders(final long contentLength,
205                                                final ContainerResponse containerResponse) {
206    final ContainerRequest containerRequest = containerResponse.getRequestContext();
207    if (containerRequest == null) {
208      throw new IllegalArgumentException("containerResponse.getRequestContext() == null");
209    }
210
211    final HttpRequest httpRequest;
212    final Object httpRequestValue = containerRequest.getProperty(HttpRequest.class.getName());
213    if (!(httpRequestValue instanceof HttpRequest)) {
214      throw new IllegalArgumentException("containerResponse; !(containerResponse.getRequestContext().getProperty(\"" +
215                                         HttpRequest.class.getName() +
216                                         "\") instanceof HttpRequest): " + httpRequestValue);
217    } else {
218      httpRequest = (HttpRequest)httpRequestValue;
219    }
220
221    final HttpVersion httpVersion = httpRequest.protocolVersion();
222    this.httpVersion = httpVersion;
223
224    final HttpResponseStatus status;
225    final StatusType responseStatusType = containerResponse.getStatusInfo();
226    if (responseStatusType == null) {
227      status = HttpResponseStatus.valueOf(containerResponse.getStatus());
228    } else {
229      final String reasonPhrase = responseStatusType.getReasonPhrase();
230      if (reasonPhrase == null) {
231        status = HttpResponseStatus.valueOf(containerResponse.getStatus());
232      } else {
233        status = HttpResponseStatus.valueOf(containerResponse.getStatus(), reasonPhrase);
234      }
235    }
236
237    final HttpMessage httpResponse;
238    final boolean needsOutputStream;
239    if (contentLength < 0L) {
240      needsOutputStream = !HttpMethod.HEAD.equalsIgnoreCase(containerRequest.getMethod());
241      httpResponse = new DefaultHttpResponse(httpVersion, status);
242      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
243      HttpUtil.setTransferEncodingChunked(httpResponse, true);
244    } else if (contentLength == 0L) {
245      needsOutputStream = false;
246      httpResponse = new DefaultFullHttpResponse(httpVersion, status);
247      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
248      HttpUtil.setContentLength(httpResponse, 0L);
249    } else {
250      needsOutputStream = !HttpMethod.HEAD.equalsIgnoreCase(containerRequest.getMethod());
251      httpResponse = new DefaultHttpResponse(httpVersion, status);
252      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
253      HttpUtil.setContentLength(httpResponse, contentLength);
254    }
255    if (HttpUtil.isKeepAlive(httpRequest)) {
256      HttpUtil.setKeepAlive(httpResponse, true);
257    }
258
259    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
260
261    final ChannelPromise channelPromise = channelHandlerContext.newPromise();
262    assert channelPromise != null;
263    channelPromise.addListener(listener);
264
265    // Remember that
266    // AbstractContainerRequestHandlingResponseWriter#channelReadComplete(ChannelHandlerContext)
267    // will call ChannelHandlerContext#flush() in all cases.
268    channelHandlerContext.write(httpResponse, channelPromise);
269
270    return needsOutputStream;
271  }
272
273  /**
274   * Creates and returns a new {@link
275   * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}.
276   *
277   * <p>This method never returns {@code null}.</p>
278   *
279   * @param contentLength the content length as determined by the
280   * logic encapsulated by the {@link
281   * ApplicationHandler#handle(ContainerRequest)} method; must not be
282   * {@code 0L}
283   *
284   * @param containerResponse a {@link ContainerResponse} for which an
285   * {@link OutputStream} is being created and returned; must not be
286   * {@code null}; ignored by this implementation
287   *
288   * @return a new {@link
289   * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}
290   *
291   * @exception NullPointerException if {@code containerResponse} is
292   * {@code null} or if {@link #getChannelHandlerContext()} returns
293   * {@code null}
294   *
295   * @exception IllegalArgumentException if {@code contentLength} is
296   * {@code 0L}
297   *
298   * @see ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
299   */
300  @Override
301  protected final AbstractChannelOutboundInvokingOutputStream<? extends HttpContent> createOutputStream(final long contentLength,
302                                                                                                        final ContainerResponse containerResponse) {
303    if (contentLength == 0L) {
304      throw new IllegalArgumentException("contentLength == 0L");
305    }
306    return new ByteBufBackedChannelOutboundInvokingHttpContentOutputStream(this.getChannelHandlerContext(),
307                                                                           this.getFlushThreshold(),
308                                                                           false,
309                                                                           this.getByteBufCreator()) {
310      @Override
311      protected final ChannelPromise newPromise() {
312        final ChannelPromise returnValue = super.newPromise();
313        if (returnValue != null && !returnValue.isVoid()) {
314          returnValue.addListener(listener);
315        }
316        return returnValue;
317      }
318    };
319  }
320
321  /**
322   * Overrides {@link
323   * AbstractContainerRequestHandlingResponseWriter#commit()} to still
324   * effectively do nothing, but clean up some internal state.
325   *
326   * @see AbstractContainerRequestHandlingResponseWriter#commit()
327   */
328  @Override
329  public final void commit() {
330    this.httpVersion = null;
331    super.commit();
332  }
333
334  /**
335   * Writes an appropriate failure message using the return value of
336   * the {@link #getChannelHandlerContext()} method.
337   *
338   * @param failureCause the {@link Throwable} responsible for this
339   * method's invocation; may be {@code null} in pathological cases;
340   * ignored by this implementation
341   *
342   * @exception NullPointerException if {@link
343   * #getChannelHandlerContext()} returns {@code null}
344   */
345  @Override
346  protected final void writeFailureMessage(final Throwable failureCause) {
347    final HttpVersion httpVersion = this.httpVersion;
348    this.httpVersion = null;
349
350    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
351
352    final ChannelPromise channelPromise = channelHandlerContext.newPromise();
353    assert channelPromise != null;
354    channelPromise.addListener(listener);
355
356    final HttpMessage failureMessage = new DefaultFullHttpResponse(httpVersion, HttpResponseStatus.INTERNAL_SERVER_ERROR);
357    HttpUtil.setContentLength(failureMessage, 0L);
358
359    channelHandlerContext.write(failureMessage, channelPromise);
360  }

367
368  private static final void copyHeaders(final Map<? extends String, ? extends List<String>> headersSource,
369                                        final HttpHeaders nettyHeaders) {
370    AbstractContainerRequestHandlingResponseWriter.copyHeaders(headersSource, UnaryOperator.identity(), nettyHeaders::add);
371  }
