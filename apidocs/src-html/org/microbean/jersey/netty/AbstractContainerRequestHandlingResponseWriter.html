055/**
056 * An abstract {@link ChannelInboundHandlerAdapter} that is also a
057 * {@link ContainerResponseWriter} that processes incoming {@link
058 * ContainerRequest} events, such as those dispatched by an {@link
059 * AbstractContainerRequestDecoder}, by supplying them to the {@link
060 * ApplicationHandler#handle(ContainerRequest)} method.
061 *
062 * <p>Instances of this class are in charge of properly invoking
063 * {@link ApplicationHandler#handle(ContainerRequest)}, thus adapting
064 * <a href="https://eclipse-ee4j.github.io/jersey/"
065 * target="_parent">Jersey</a> to <a href="https://netty.io/"
066 * target="_parent">Netty</a>'s constraints and vice versa.</p>
067 *
068 * @param <T> the type of message that will be written by instances of
069 * this class; see {@link #createOutputStream(long,
070 * ContainerResponse)}
071 *
072 * @author <a href="https://about.me/lairdnelson"
073 * target="_parent">Laird Nelson</a>
074 *
075 * @see #channelRead(ChannelHandlerContext, Object)
076 *
077 * @see #createOutputStream(long, ContainerResponse)
078 * 
079 * @see ChannelInboundHandlerAdapter
080 *
081 * @see ApplicationHandler#handle(ContainerRequest)
082 *
083 * @see ContainerResponseWriter
084 */
085public abstract class AbstractContainerRequestHandlingResponseWriter<T> extends ChannelInboundHandlerAdapter implements ContainerResponseWriter {

210  /**
211   * Overrides {@link
212   * ChannelInboundHandlerAdapter#channelActive(ChannelHandlerContext)}
213   * to ensure that a channel read is performed one way or another.
214   *
215   * <p>If {@linkplain ChannelConfig#isAutoRead() autoread is active},
216   * then the superclass' version of this method is executed with no
217   * changes.  If {@linkplain ChannelConfig#isAutoRead() autoread is
218   * inactive}, then a call is made to {@link
219   * ChannelHandlerContext#read()}.</p>
220   *
221   * @param channelHandlerContext the {@link ChannelHandlerContext} in
222   * effect; must not be {@code null}
223   *
224   * @exception NullPointerException if {@code channelHandlerContext}
225   * is {@code null}
226   *
227   * @exception Exception if an error occurs
228   *
229   * @see ChannelConfig#isAutoRead()
230   *
231   * @see Channel#read()
232   *
233   * @see ChannelHandlerContext#read()
234   *
235   * @see #channelRead(ChannelHandlerContext, Object)
236   */
237  @Override
238  public final void channelActive(final ChannelHandlerContext channelHandlerContext) throws Exception {
239    super.channelActive(channelHandlerContext);
240    // See
241    // https://github.com/netty/netty/blob/d446765b8469ca40db40f46e5c637d980b734a8a/transport/src/main/java/io/netty/channel/DefaultChannelPipeline.java#L1408-L1413.
242    if (!channelHandlerContext.channel().config().isAutoRead()) {
243      // If autoRead was "on", then the superclass version of this
244      // method will have already performed a read from the channel
245      // (e.g. channel.read()).  If autoRead is "off", then we get
246      // here, and we do a read from the ChannelHandlerContext (and so
247      // we don't have to traverse the whole pipeline).
248      channelHandlerContext.read();
249    }
250  }

252  /**
253   * If the supplied {@code message} is a {@link ContainerRequest}
254   * then this method will {@linkplain
255   * ContainerRequest#setWriter(ContainerResponseWriter) install
256   * itself as that request's <code>ContainerResponseWriter</code>}
257   * and will invoke {@link
258   * ApplicationHandler#handle(ContainerRequest)}.
259   *
260   * <p>In all other cases this method will simply call {@link
261   * ChannelInboundHandlerAdapter#channelRead(ChannelHandlerContext,
262   * Object)} with the supplied {@code message}.</p>
263   *
264   * @param channelHandlerContext the {@link ChannelHandlerContext} in
265   * effect; must not be {@code null}
266   *
267   * @param message the incoming message, or event; may be {@code null}
268   *
269   * @exception NullPointerException if {@code channelHandlerContext}
270   * is {@code null}
271   *
272   * @exception Exception if {@code message} is not an instance of
273   * {@link ContainerRequest} and {@link
274   * ChannelInboundHandlerAdapter#channelRead(ChannelHandlerContext,
275   * Object)} throws an {@link Exception}
276   *
277   * @see ApplicationHandler#handle(ContainerRequest)
278   *
279   * @see ContainerResponseWriter
280   *
281   * @see ContainerRequest#setWriter(ContainerResponseWriter)
282   */
283  @Override
284  public final void channelRead(final ChannelHandlerContext channelHandlerContext,
285                                final Object message)
286    throws Exception {
287    if (this.getChannelHandlerContext() != null) {
288      throw new IllegalStateException("this.getChannelHandlerContext() != null: " + this.getChannelHandlerContext());
289    }
290    this.channelHandlerContext = Objects.requireNonNull(channelHandlerContext);
291    try {
292      if (message instanceof ContainerRequest) {
293        final ContainerRequest containerRequest = (ContainerRequest)message;
294        containerRequest.setWriter(this);
295        this.applicationHandler.handle(containerRequest);
296      } else {
297        super.channelRead(channelHandlerContext, message);
298      }
299    } finally {
300      this.channelHandlerContext = null;
301    }
302  }

304  /**
305   * Overrides the {@link
306   * ChannelInboundHandlerAdapter#channelReadComplete(ChannelHandlerContext)}
307   * method to call {@link ChannelHandlerContext#flush()
308   * channelHandlerContext.flush()} before calling the superclass
309   * implementation.
310   *
311   * @param channelHandlerContext the {@link ChannelHandlerContext} in
312   * effect; must not be {@code null}
313   *
314   * @exception NullPointerException if {@code channelHandlerContext} is {@code null}
315   *
316   * @exception Exception if {@link
317   * ChannelInboundHandlerAdapter#channelReadComplete(ChannelHandlerContext)}
318   * throws an {@link Exception}
319   */
320  @Override
321  public final void channelReadComplete(final ChannelHandlerContext channelHandlerContext) throws Exception {
322    channelHandlerContext.flush();
323    // See
324    // https://github.com/netty/netty/blob/d446765b8469ca40db40f46e5c637d980b734a8a/transport/src/main/java/io/netty/channel/DefaultChannelPipeline.java#L1408-L1413.
325    super.channelReadComplete(channelHandlerContext);
326    if (!channelHandlerContext.channel().config().isAutoRead()) {
327      // Ultimately a read is just an idempotent call so even if other
328      // handlers do this seemingly nothing bad will happen.  See
329      // https://github.com/netty/netty/blob/9976ab7fe86e052d29ca7accf528c885e93dcb4c/transport/src/main/java/io/netty/channel/nio/AbstractNioChannel.java#L402-L416.
330      // Even in the ancient case of blocking IO it's still
331      // idempotent:
332      // https://github.com/netty/netty/blob/9976ab7fe86e052d29ca7accf528c885e93dcb4c/transport/src/main/java/io/netty/channel/oio/AbstractOioChannel.java#L101-L109
333      // Finally, transports like Epoll are less clear:
334      // https://github.com/netty/netty/blob/9976ab7fe86e052d29ca7accf528c885e93dcb4c/transport-native-epoll/src/main/java/io/netty/channel/epoll/AbstractEpollChannel.java#L226-L242
335      // ...but Epoll too ultimately is idempotent:
336      // https://github.com/netty/netty/blob/9976ab7fe86e052d29ca7accf528c885e93dcb4c/transport-native-epoll/src/main/java/io/netty/channel/epoll/AbstractEpollChannel.java#L226-L242
337      channelHandlerContext.read();
338    }
339  }

443    final OutputStream returnValue;
444    if (this.writeStatusAndHeaders(contentLength, Objects.requireNonNull(containerResponse)) && contentLength != 0L) {

446    } else {
447      returnValue = null;
448    }
449    return returnValue;
450  }

609  @Override
610  public final boolean suspend(final long timeout,
611                               final TimeUnit timeUnit,
612                               final TimeoutHandler timeoutHandler) {
613    // Lifted from Jersey's supplied Netty integration, with repairs.
614    final boolean returnValue;
615    if (timeoutHandler == null || this.suspendTimeoutHandler != null) {
616      returnValue = false;
617    } else {
618      this.suspendTimeoutHandler = () -> {
619        timeoutHandler.onTimeout(this);
620        this.suspendTimeoutHandler = null;
621      };
622      if (timeout > 0L) {
623        final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
624        this.suspendTimeoutFuture =
625          channelHandlerContext.executor().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
626      }
627      returnValue = true;
628    }
629    return returnValue;
630  }

632  @Override
633  public final void setSuspendTimeout(final long timeout, final TimeUnit timeUnit) {
634    // Lifted from Jersey's supplied Netty integration, with repairs.
635    if (this.suspendTimeoutHandler == null) {
636      throw new IllegalStateException("this.suspendTimeoutHandler == null");
637    }
638    if (this.suspendTimeoutFuture != null) {
639      this.suspendTimeoutFuture.cancel(true);
640      this.suspendTimeoutFuture = null;
641    }
642    if (timeout > 0L) {
643      final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
644      this.suspendTimeoutFuture =
645        channelHandlerContext.executor().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
646    }
647  }

649  /**
650   * Handles any failure case encountered by the logic encapsulated by
651   * the {@link ApplicationHandler#handle(ContainerRequest)} method.
652   *
653   * <p>Either {@link #commit()} or {@link #failure(Throwable)} will
654   * be called by Jersey as the logically final operation in the logic
655   * encapsulated by the {@link
656   * ApplicationHandler#handle(ContainerRequest)} method, but not
657   * both.</p>
658   *
659   * <p>This method calls the {@link #writeFailureMessage(Throwable)}
660   * method and takes great care to ensure that any {@link Throwable}s
661   * encountered along the way are properly recorded and {@linkplain
662   * Throwable#addSuppressed(Throwable) suppressed}.</p>
663   *
664   * <p><strong>This implementation never returns.</strong> A {@link
665   * ContainerException} is always thrown by this method.</p>
666   *
667   * @param failureCause the {@link Throwable} encountered by the
668   * {@link ApplicationHandler#handle(ContainerRequest)} method; may
669   * be {@code null}
670   *
671   * @exception ContainerException when this method is invoked; it
672   * will have the supplied {@code failureCause} as its {@linkplain
673   * Throwable#getCause() cause}
674   *
675   * @see #commit()
676   *
677   * @see ContainerResponseWriter#failure(Throwable)
678   */
679  @Override
680  public final void failure(final Throwable failureCause) {
681    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
682    Throwable outerWriteProblem = null;
683    try {
684      this.writeFailureMessage(failureCause);
685    } catch (final RuntimeException | Error writeProblem) {
686      outerWriteProblem = writeProblem;
687      if (failureCause != null) {
688        boolean foundFailureCauseInSuppressedThrowables = false;
689        final Object[] suppressedThrowables = writeProblem.getSuppressed();
690        if (suppressedThrowables != null && suppressedThrowables.length > 0) {
691          for (final Object suppressedThrowable : suppressedThrowables) {
692            if (suppressedThrowable == failureCause) {
693              foundFailureCauseInSuppressedThrowables = true;
694              break;
695            }
696          }
697        }
698        if (!foundFailureCauseInSuppressedThrowables) {
699          writeProblem.addSuppressed(failureCause);
700        }
701      }
702      throw writeProblem;
703    } finally {
704      Throwable outerFlushProblem = null;
705      try {
706        channelHandlerContext.flush();
707      } catch (final RuntimeException | Error flushProblem) {
708        outerFlushProblem = flushProblem;
709        if (outerWriteProblem != null) {
710          flushProblem.addSuppressed(outerWriteProblem);
711        } else if (failureCause != null) {
712          flushProblem.addSuppressed(failureCause);
713        }
714        throw flushProblem;
715      } finally {
716        try {
717          channelHandlerContext.close();
718        } catch (final RuntimeException | Error closeProblem) {
719          if (outerFlushProblem != null) {
720            closeProblem.addSuppressed(outerFlushProblem);
721          } else if (failureCause != null) {
722            closeProblem.addSuppressed(failureCause);
723          }
724          throw closeProblem;
725        }
726      }
727    }
728    if (failureCause == null) {
729      throw new ContainerException("failure");
730    } else if (failureCause instanceof RuntimeException) {
731      throw (RuntimeException)failureCause;
732    } else if (failureCause instanceof Exception) {
733      throw new ContainerException(failureCause.getMessage(), failureCause);
734    } else {
735      throw (Error)failureCause;
736    }
737  }

746   * @param failureCause the {@link Throwable} responsible for this
747   * method's invocation; may be {@code null} in pathological cases

749  protected abstract void writeFailureMessage(final Throwable failureCause);

757  /**
758   * A utility function that copies entries from a source {@link Map}
759   * by passing each entry to the supplied {@link BiConsumer},
760   * transforming the keys beforehand using the supplied {@link
761   * UnaryOperator} and that is intended in this framework to be used
762   * to copy HTTP or HTTP/2 headers to and from the proper places.
763   *
764   * @param headersSource the source of the headers to copy; may be
765   * {@code null} in which case no action will be taken
766   *
767   * @param keyTransformer a {@link UnaryOperator} that transforms a
768   * header name; if {@code null} then the return value of {@link
769   * UnaryOperator#identity()} will be used instead
770   *
771   * @param headersTarget where the headers will be copied to; may be
772   * {@code null} in which case no action will be taken
773   */
774  protected static final void copyHeaders(final Map<? extends String, ? extends List<String>> headersSource,
775                                          UnaryOperator<String> keyTransformer,
776                                          final BiConsumer<? super String, ? super List<String>> headersTarget) {
777    if (headersTarget != null && headersSource != null && !headersSource.isEmpty()) {
778      final Collection<? extends Entry<? extends String, ? extends List<String>>> entrySet = headersSource.entrySet();
779      if (entrySet != null && !entrySet.isEmpty()) {
780        if (keyTransformer == null) {
781          keyTransformer = UnaryOperator.identity();
782        }
783        for (final Entry<? extends String, ? extends List<String>> entry : entrySet) {
784          if (entry != null) {
785            headersTarget.accept(keyTransformer.apply(entry.getKey()), entry.getValue());
786          }
787        }
788      }
789    }
790  }
