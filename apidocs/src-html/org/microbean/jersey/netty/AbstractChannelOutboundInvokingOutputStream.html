032/**
033 * An {@link OutputStream} that delegates writing and flushing
034 * operations to a {@link ChannelOutboundInvoker}.
035 *
036 * <h2>Thread Safety</h2>
037 *
038 * <p>Instances of this class are safe for concurrent use by multiple
039 * threads.</p>
040 *
041 * @param <T> the type of message that will be written
042 *
043 * @author <a href="https://about.me/lairdnelson"
044 * target="_parent">Laird Nelson</a>
045 *
046 * @see ChannelOutboundInvoker
047 */
048public abstract class AbstractChannelOutboundInvokingOutputStream<T> extends OutputStream {
049

121  /**
122   * Creates a new {@link
123   * AbstractChannelOutboundInvokingOutputStream}.
124   *
125   * @param channelOutboundInvoker the {@link ChannelOutboundInvoker}
126   * to which operations are adapted; must not be {@code null}
127   *
128   * @param flushThreshold the minimum number of bytes that this
129   * instance has to {@linkplain #write(byte[], int, int) write}
130   * before an automatic {@linkplain #flush() flush} will take place;
131   * if less than {@code 0} {@code 0} will be used instead; if {@code
132   * Integer#MAX_VALUE} then no automatic flushing will occur
133   *
134   * @param closeChannelOutboundInvoker whether {@link
135   * ChannelOutboundInvoker#close(ChannelPromise)} will be called on
136   * the supplied {@link ChannelOutboundInvoker} when {@link #close()
137   * close()} is called
138   *
139   * @exception NullPointerException if {@code channelOutboundInvoker}
140   * is {@code null}
141   *
142   * @see ChannelOutboundInvoker
143   *
144   * @see #getFlushThreshold()
145   *
146   * @see #close()
147   */
148  protected AbstractChannelOutboundInvokingOutputStream(final ChannelOutboundInvoker channelOutboundInvoker,
149                                                        final int flushThreshold,
150                                                        final boolean closeChannelOutboundInvoker) {
151    super();
152    this.flushThreshold = Math.max(0, flushThreshold);
153    this.channelOutboundInvoker = Objects.requireNonNull(channelOutboundInvoker);
154    this.closeChannelOutboundInvoker = closeChannelOutboundInvoker;
155  }

163  /**
164   * Returns the minimum number of bytes that this {@link
165   * AbstractChannelOutboundInvokingOutputStream} implementation has
166   * to {@linkplain #write(byte[], int, int) write} before an
167   * automatic {@linkplain #flush() flush} will take place.
168   *
169   * <p>This method will always return {@code 0} or a positive {@code
170   * int}.</p>
171   *
172   * <p>If this method returns {@code 0}, then a call to {@link
173   * #flush()} will be made at some point after every {@link
174   * #write(byte[], int, int)} invocation.</p>
175   *
176   * <p>If this method returns {@link Integer#MAX_VALUE}, then no
177   * automatic flushing will occur.</p>
178   *
179   * @return the minimum number of bytes that this {@link
180   * AbstractChannelOutboundInvokingOutputStream} implementation has
181   * to {@linkplain #write(byte[], int, int) write} before an
182   * automatic {@linkplain #flush() flush} will take place; always
183   * {@code 0} or a positive {@code int}
184   *
185   * @see
186   * #AbstractChannelOutboundInvokingOutputStream(ChannelOutboundInvoker,
187   * int, boolean)
188   *
189   * @see ChannelOutboundInvoker#flush()
190   */
191  public final int getFlushThreshold() {
192    return this.flushThreshold;
193  }
194
195  @Override
196  public final void write(final int singleByte) throws IOException {
197    this.write(this.createMessage(singleByte), 1);
198  }
199
200  @Override
201  public final void write(final byte[] bytes) throws IOException {
202    this.write(bytes, 0, bytes.length);
203  }
204
205  @Override
206  public final void write(final byte[] bytes, final int offset, final int length) throws IOException {
207    if (offset < 0 || length < 0 || offset + length > bytes.length) {
208      throw new IndexOutOfBoundsException();
209    }
210    this.write(this.createMessage(bytes, offset, length), length);
211  }
212
213  private final void write(final T message, final int length) {
214    final ChannelPromise channelPromise = this.newPromise();
215    final int flushThreshold = this.getFlushThreshold();
216    switch (flushThreshold) {
217    case 0:
218      // Flush previous writes, if any
219      this.channelOutboundInvoker.flush();
220      break;
221    case Integer.MAX_VALUE:
222      break;
223    default:
224      int bytesWritten = this.bytesWritten; // volatile read
225      if (bytesWritten > flushThreshold) {
226        // Flush previous writes, if any, and set our "days since flush" back to 0 (see #flush())
227        this.flush();
228      } else if (channelPromise.isVoid()) {
229        // Optimistically assume the write will succeed; if we get
230        // this wrong, all that happens is maybe we don't flush as
231        // often as expected.
232        this.bytesWritten = bytesWritten + length;
233      } else {
234        channelPromise.addListener(f -> {
235            if (f.isSuccess()) {
236              this.bytesWritten = bytesWritten + length;
237            }
238          });
239      }
240    }
241    this.channelOutboundInvoker.write(message, channelPromise);
242  }

244  /**
245   * Returns a new message representing the single supplied {@code
246   * byte} to be {@linkplain ChannelOutboundInvoker#write(Object,
247   * ChannelPromise) written} by this {@link
248   * AbstractChannelOutboundInvokingOutputStream}'s various {@link
249   * #write(byte[], int, int) write} methods.
250   *
251   * <p>This method never returns {@code null}.</p>
252   *
253   * <p>Overrides of this method must not return {@code null}.</p>
254   *
255   * <p>The default implementation of this method is inefficient: it
256   * creates a new {@code byte[]} holding the sole {@code byte}
257   * represented by the {@code singleByte} parameter value and calls
258   * {@link #createMessage(byte[], int, int)} and returns its result.
259   * Subclasses are encouraged, but not required, to override this
260   * method to be more efficient.</p>
261   *
262   * <p>Overrides of this method should be stateless.</p>
263   *
264   * @param singleByte an {@code int} whose low-order bits hold a
265   * {@code byte} to be written
266   *
267   * @return a new, non-{@code null} message to {@linkplain
268   * ChannelOutboundInvoker#write(Object, ChannelPromise) write}
269   *
270   * @exception IOException if an error occurs
271   *
272   * @see #createMessage(byte[], int, int)
273   *
274   * @see ChannelOutboundInvoker#write(Object, ChannelPromise)
275   */
276  protected T createMessage(final int singleByte) throws IOException {
277    return this.createMessage(new byte[] { (byte)singleByte }, 0, 1);
278  }

341   * Creates and returns new {@link ChannelPromise}s that will be used

347   *
348   * <p>The default implementation of this method returns the return
349   * value of invoking {@link ChannelOutboundInvoker#newPromise()}.
350   *
351   * @return a new, non-{@code null} {@link ChannelPromise} that will
352   * be supplied to many {@link ChannelOutboundInvoker} operations

355   *
356   * @see ChannelOutboundInvoker#newPromise()
357   *

361    return this.channelOutboundInvoker.newPromise();
362  }

364  /**
365   * Calls the {@link ChannelOutboundInvoker#flush()} method on the
366   * {@link ChannelOutboundInvoker} {@linkplain
367   * #AbstractChannelOutboundInvokingOutputStream(ChannelOutboundInvoker,
368   * int, boolean) supplied at construction time}.
369   *
370   * @see ChannelOutboundInvoker#flush()
371   *
372   * @see #getFlushThreshold()
373   *
374   * @see
375   * #AbstractChannelOutboundInvokingOutputStream(ChannelOutboundInvoker,
376   * int, boolean)
377   */
378  @Override
379  public final void flush() {
380    this.channelOutboundInvoker.flush();
381    this.bytesWritten = 0;
382  }
383
384  /**
385   * {@linkplain OutputStream#close() Closes} this {@link
386   * AbstractChannelOutboundInvokingOutputStream}, optionally
387   * {@linkplain ChannelOutboundInvoker#writeAndFlush(Object,
388   * ChannelPromise) writing and flushing} a {@linkplain
389   * #createLastMessage() final message}, or simply just {@linkplain
390   * #flush() flushing} first, before possibly {@linkplain
391   * ChannelOutboundInvoker#close(ChannelPromise) closing the
392   * underlying <code>ChannelOutboundInvoker</code>}.
393   *
394   * @exception IOException if the {@link #createLastMessage()} method
395   * throws an {@link IOException}
396   *
397   * @see #createLastMessage()
398   *
399   * @see ChannelOutboundInvoker#close(ChannelPromise)
400   *
401   * @see
402   * #AbstractChannelOutboundInvokingOutputStream(ChannelOutboundInvoker,
403   * int, boolean)
404   */
405  @Override
406  public final void close() throws IOException {
407    super.close();
408    final Object lastMessage = this.createLastMessage();
409    if (lastMessage == null) {
410      this.flush();
411    } else {
412      final ChannelPromise channelPromise = this.newPromise();
413      if (channelPromise.isVoid()) {
414        this.channelOutboundInvoker.writeAndFlush(lastMessage, channelPromise);
415        this.bytesWritten = 0;
416      } else {
417        channelPromise.addListener(f -> {
418            if (f.isSuccess()) {
419              this.bytesWritten = 0;
420            }
421          });
422        this.channelOutboundInvoker.writeAndFlush(lastMessage, channelPromise);
423      }
424    }
425    if (this.closeChannelOutboundInvoker) {
426      this.channelOutboundInvoker.close(this.newPromise());
427    }
428  }
