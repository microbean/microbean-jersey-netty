018
019import java.net.URI;
020
021import java.util.List; // for javadoc only
022
023import javax.ws.rs.core.Configuration;
024
025import io.netty.buffer.ByteBuf;
026import io.netty.buffer.ByteBufHolder;
027
028import io.netty.channel.ChannelHandlerContext; // for javadoc only
029
030import io.netty.handler.codec.http.FullHttpMessage;
031import io.netty.handler.codec.http.HttpContent;
032import io.netty.handler.codec.http.HttpMessage;
033import io.netty.handler.codec.http.HttpObject;
034import io.netty.handler.codec.http.HttpRequest;

036import io.netty.handler.codec.http.LastHttpContent;
037
038import org.glassfish.jersey.server.ContainerRequest;

040/**
041 * An {@link AbstractContainerRequestDecoder} that {@linkplain
042 * #decode(ChannelHandlerContext, Object, List) decodes} {@link
043 * HttpObject}s into {@link ContainerRequest}s.
044 *
045 * @author <a href="https://about.me/lairdnelson"
046 * target="_parent">Laird Nelson</a>
047 *
048 * @see #decode(ChannelHandlerContext, Object, List)
049 */
050public final class HttpObjectToContainerRequestDecoder extends AbstractContainerRequestDecoder<HttpObject, HttpRequest, HttpContent> {
051
052
053  /*
054   * Constructors.
055   */
056
057
058  /**
059   * Creates a new {@link HttpObjectToContainerRequestDecoder}.
060   *
061   * @param baseUri a {@link URI} that will serve as the {@linkplain
062   * ContainerRequest#getBaseUri() base <code>URI</code>} in a new
063   * {@link ContainerRequest}; may be {@code null} in which case the
064   * return value of {@link URI#create(String) URI.create("/")} will
065   * be used instead
066   *
067   * @see #HttpObjectToContainerRequestDecoder(URI, Configuration)
068   *
069   * @deprecated Please use the {@link
070   * #HttpObjectToContainerRequestDecoder(URI, Configuration)}
071   * constructor instead.
072   */
073  @Deprecated
074  public HttpObjectToContainerRequestDecoder(final URI baseUri) {
075    this(baseUri, null);
076  }
077
078  /**
079   * Creates a new {@link HttpObjectToContainerRequestDecoder}.
080   *
081   * @param baseUri a {@link URI} that will serve as the {@linkplain
082   * ContainerRequest#getBaseUri() base <code>URI</code>} in a new
083   * {@link ContainerRequest}; may be {@code null} in which case the
084   * return value of {@link URI#create(String) URI.create("/")} will
085   * be used instead
086   *
087   * @param configuration a {@link Configuration} describing how the
088   * container is configured; may be {@code null}
089   */
090  public HttpObjectToContainerRequestDecoder(final URI baseUri, final Configuration configuration) {
091    super(baseUri, configuration, HttpRequest.class, HttpContent.class);
092  }
093
094
095  /*
096   * Instance methods.
097   */
098  
099
100  /**
101   * Extracts and returns a {@link String} representing a request URI
102   * from the supplied message, which is guaranteed to be a
103   * {@linkplain #isHeaders(Object) "headers" message}.
104   *
105   * <p>This implementation calls {@link HttpRequest#uri()
106   * httpRequest.uri()} and returns the result.</p>
107   *
108   * @param httpRequest the message to interrogate; will not be {@code
109   * null}
110   *
111   * @return a {@link String} representing a request URI from the
112   * supplied message, or {@code null}
113   */
114  @Override
115  protected final String getRequestUriString(final HttpRequest httpRequest) {
116    return httpRequest.uri();
117  }
118
119  /**
120   * Extracts and returns the name of the request method from the
121   * supplied message, which is guaranteed to be a {@linkplain
122   * #isHeaders(Object) "headers" message}.
123   *
124   * <p>This implementation calls {@link HttpRequest#method()
125   * httpRequest.method().name()} and returns the result.</p>
126   *
127   * @param httpRequest the message to interrogate; will not be {@code
128   * null}
129   *
130   * @return a {@link String} representing the request method from the
131   * supplied message, or {@code null}
132   */
133  @Override
134  protected final String getMethod(final HttpRequest httpRequest) {
135    return httpRequest.method().name();
136  }
137
138  /**
139   * Returns {@code true} if the supplied {@link HttpObject} is the
140   * last of a stream of messages.
141   *
142   * <p>This implementation returns {@code true} if either:</p>
143   *
144   * <ul>
145   *
146   * <li>{@code httpObject} is an instance of {@link LastHttpContent},
147   * or</li>
148   *
149   * <li>{@code httpObject} is an instance of {@link HttpRequest} and
150   * its {@linkplain HttpUtil#getContentLength(HttpMessage, long)
151   * content length} equals {@code 0L}</li>
152   *
153   * </ul>
154   *
155   * @param httpObject the message to interrogate; will not be {@code
156   * null}
157   *
158   * @return {@code true} if no further messages in the stream are
159   * forthcoming; {@code false} otherwise
160   */
161  @Override
162  protected final boolean isLast(final HttpObject httpObject) {
163    return httpObject instanceof LastHttpContent || (httpObject instanceof HttpRequest && HttpUtil.getContentLength((HttpRequest)httpObject, -1L) == 0L);
164  }
165
166}
